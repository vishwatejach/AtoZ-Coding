void bubbleSort(vector<int> &arr, int n) 
{
    //write your code here
    for(int i=n-1; i>=0; i--)
    {
        int done = 0;
        for(int j=0; j<i; j++)
        {
            if(arr[j]>arr[j+1])
            {
                swap(arr[j],arr[j+1]);
                done = 1;
            }
        }
        if(done == 0) break;
    }
}
